        function hexToRgb(hex) {
            let result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
            return result ? { r: parseInt(result[1], 16), g: parseInt(result[2], 16), b: parseInt(result[3], 16) } : null;

        function handleScrollColorChange() {
            const colorThemes = [{primary:'#6a11cb',secondary:'#2575fc'},{primary:'#ef476f',secondary:'#ffd166'},{primary:'#06d6a0',secondary:'#118ab2'},{primary:'#d62828',secondary:'#f77f00'},{primary:'#43cea2',secondary:'#185a9d'}];
            const root = document.documentElement;
            let currentThemeIndex = -1;
            
            return function() {
                const scrollableHeight = document.documentElement.scrollHeight - window.innerHeight;
                if (scrollableHeight <= 0) return;
                const scrollPercent = (window.scrollY / scrollableHeight) * 100;
                const themeChangeThreshold = 100 / colorThemes.length;
                const newThemeIndex = Math.min(Math.floor(scrollPercent / themeChangeThreshold), colorThemes.length - 1);

                if (newThemeIndex !== currentThemeIndex) {
                    currentThemeIndex = newThemeIndex;
                    const newTheme = colorThemes[currentThemeIndex];
                    root.style.setProperty('--primary-color', newTheme.primary);
                    root.style.setProperty('--secondary-color', newTheme.secondary);
                    const cardRgb = hexToRgb(newTheme.primary);
                    if (cardRgb) {
                        root.style.setProperty('--card-bg-color', `rgba(${cardRgb.r}, ${cardRgb.g}, ${cardRgb.b}, 0.05)`);
                    }
                }
            };
        }
