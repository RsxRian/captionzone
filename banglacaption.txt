        // গ্লোবাল ভেরিয়েবল
        const captionGrid = document.getElementById('caption-grid');
        const filterButtons = document.querySelectorAll('.filter-btn');
        const AD_INTERVAL = 5; // প্রতি ৫টি ক্যাপশনের পর বিজ্ঞাপন

        // =============================================
        // মূল ফাংশনসমূহ
        // =============================================
        
        /**
         * ক্যাপশন এবং বিজ্ঞাপন মিশ্রিত করে একটি নতুন তালিকা তৈরি করে
         * @param {Array} captionsList - ফিল্টার করা ক্যাপশনের তালিকা
         * @returns {Array} - বিজ্ঞাপনসহ নতুন মিশ্রিত তালিকা
         */
        function createCombinedList(captionsList) {
            const combined = [];
            if (myAds.length === 0) return captionsList;

            let adCounter = 0;
            for (let i = 0; i < captionsList.length; i++) {
                combined.push(captionsList[i]);
                if ((i + 1) % AD_INTERVAL === 0) {
                    const adData = myAds[adCounter % myAds.length];
                    combined.push({ ...adData, isAd: true });
                    adCounter++;
                }
            }
            return combined;
        }

        /**
         * গ্রিডে কার্ডগুলো (ক্যাপশন বা বিজ্ঞাপন) রেন্ডার করে
         * @param {Array} itemsToRender - প্রদর্শন করার জন্য আইটেমের তালিকা
         */
        function renderGridItems(itemsToRender) {
            captionGrid.innerHTML = ''; 
            if (itemsToRender.length === 0) {
                captionGrid.innerHTML = `<p style="text-align:center; grid-column: 1/-1;">এই ক্যাটাগরিতে কোনো ক্যাপশন পাওয়া যায়নি।</p>`;
                return;
            }

            itemsToRender.forEach(item => {
                let card;
                if (item.isAd) {
                    card = document.createElement('div'); 
                    card.classList.add('ad-card');
                    card.innerHTML = `<div class="ad-tag">বিজ্ঞাপন</div><h3 class="ad-title">${item.title}</h3><p class="ad-text">${item.text}</p><a href="${item.link}" target="_blank" class="ad-button">${item.cta} <i class="fa-solid fa-arrow-right-long"></i></a>`;
                } else {
                    card = document.createElement('div'); 
                    card.classList.add('caption-card');
                    card.innerHTML = `<p class="caption-text">${item.text}</p><div class="card-footer"><button class="copy-btn" data-caption="${item.text}"><i class="fa-regular fa-copy"></i> কপি করুন</button></div>`;
                }
                captionGrid.appendChild(card);
            });
        }

        /**
         * হেক্স কালারকে RGB তে রূপান্তর করে
         */
        function hexToRgb(hex) {
            let result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
            return result ? { r: parseInt(result[1], 16), g: parseInt(result[2], 16), b: parseInt(result[3], 16) } : null;
        }
        
        /**
         * স্ক্রল করার সময় থিমের রঙ পরিবর্তন করে
         */
        function handleScrollColorChange() {
            const colorThemes = [{primary:'#6a11cb',secondary:'#2575fc'},{primary:'#ef476f',secondary:'#ffd166'},{primary:'#06d6a0',secondary:'#118ab2'},{primary:'#d62828',secondary:'#f77f00'},{primary:'#43cea2',secondary:'#185a9d'}];
            const root = document.documentElement;
            let currentThemeIndex = -1;
            
            return function() {
                const scrollableHeight = document.documentElement.scrollHeight - window.innerHeight;
                if (scrollableHeight <= 0) return;
                const scrollPercent = (window.scrollY / scrollableHeight) * 100;
                const themeChangeThreshold = 100 / colorThemes.length;
                const newThemeIndex = Math.min(Math.floor(scrollPercent / themeChangeThreshold), colorThemes.length - 1);

                if (newThemeIndex !== currentThemeIndex) {
                    currentThemeIndex = newThemeIndex;
                    const newTheme = colorThemes[currentThemeIndex];
                    root.style.setProperty('--primary-color', newTheme.primary);
                    root.style.setProperty('--secondary-color', newTheme.secondary);
                    const cardRgb = hexToRgb(newTheme.primary);
                    if (cardRgb) {
                        root.style.setProperty('--card-bg-color', `rgba(${cardRgb.r}, ${cardRgb.g}, ${cardRgb.b}, 0.05)`);
                    }
                }
            };
        }

        /**
         * ক্লিপবোর্ডে টেক্সট কপি করার ফাংশন
         */
        function copyCaption(text, button) {
            function fallbackCopy(text) { 
                const textArea = document.createElement("textarea");
                textArea.value = text;
                textArea.style.position="fixed";
                textArea.style.top="-9999px";
                document.body.appendChild(textArea);
                textArea.focus();
                textArea.select();
                try { return document.execCommand('copy'); } finally { document.body.removeChild(textArea); }
            }
            function showSuccess(button) { 
                button.innerHTML=`<i class="fa-solid fa-check"></i> কপি হয়েছে!`;
                button.classList.add('copied');
                setTimeout(() => {
                    button.innerHTML=`<i class="fa-regular fa-copy"></i> কপি করুন`;
                    button.classList.remove('copied');
                }, 2000); 
            }
            
            (navigator.clipboard && window.isSecureContext ? navigator.clipboard.writeText(text) : Promise.reject())
            .catch(() => fallbackCopy(text) ? Promise.resolve() : Promise.reject())
            .then(() => showSuccess(button))
            .catch(() => alert("কপি করা যায়নি।"));
        }

        // ক্যাটাগরি ফিল্টার বাটন
        filterButtons.forEach(button => { 
            button.addEventListener('click', () => { 
                filterButtons.forEach(btn => btn.classList.remove('active')); 
                button.classList.add('active'); 
                
                const category = button.dataset.category;
                const filteredCaptions = category === 'all' ? captions : captions.filter(c => c.category === category); 
                const combinedItems = createCombinedList(filteredCaptions);
                renderGridItems(combinedItems); 
            }); 
        });

        captionGrid.addEventListener('click', function(e) { 
            const button = e.target.closest('.copy-btn'); 
            if (button && !button.classList.contains('copied')) { 
                copyCaption(button.dataset.caption, button); 
            } 
        });

        // স্ক্রল ইভেন্ট
        const scrollHandler = handleScrollColorChange();

        window.addEventListener('DOMContentLoaded', () => {

        });
